		limit = {
			FROM.FROM = {
				state = 335 #Batavia
			}
		}

			limit = {

					OR = {
						tag = JAP
						tag = IDN
					}
				}
			}
			set_province_name = {
				id = 7381
				name = VICTORY_POINTS_7381_INS
			}

				id = 4406
				name = VICTORY_POINTS_4406_INS

			FROM.FROM = {
				set_state_name = STATE_335_INS
			}
		}

			limit = {
				FROM = { tag = DEI }
			}
			FROM.FROM = { reset_state_name = yes }
			reset_province_name = 7381
			reset_province_name = 4406
		}
